24 void update(const SDL_Event &event) override {
25 if (m_status == ButtonStatus::Disabled) {
26 return;
27 }
28 float x = 0, y = 0;
29 if (event.type == SDL_EVENT_MOUSE_MOTION) {
30 x = event.motion.x;
31 y = event.motion.y;
32 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN ||
33 event.type == SDL_EVENT_MOUSE_BUTTON_UP) {
34 x = event.button.x;
35 y = event.button.y;
36 } else {
37 return;
38 }
39 if (!m_boundingBox.isInBoundingBox(x, y)) {
40 m_status = ButtonStatus::Normal;
41 return;
42 }
43 if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN &&
44 event.button.button == SDL_BUTTON_LEFT) {
45 m_status = ButtonStatus::Pressed;
46 return;
47 }
48 if (event.type == SDL_EVENT_MOUSE_BUTTON_UP &&
49 event.button.button == SDL_BUTTON_LEFT &&
50 m_status == ButtonStatus::Pressed) {
51 m_status = ButtonStatus::Hover;
52 m_onClick();
53 return;
54 }
55 m_status = ButtonStatus::Hover;
56 }
57
58
59 [[nodiscard]] ButtonStatus getStatus() const override {
60 return m_status;
61 }
