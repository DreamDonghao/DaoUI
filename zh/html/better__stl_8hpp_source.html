56 /// @brief 可控删除器
57 /// @details 可以为std::unique_ptr 提供一个删除器，使其可以有观察者与管理者两种状态
58 template<typename T>
59 class SwitchDeleter {
60 public:
61 enum Mode { Observe, Manage };
62
63 explicit SwitchDeleter(const Mode m = Manage) : mode(m) {
64 }
65
66 void operator()(const T *ptr) const noexcept {
67 if (mode == Manage) {
68 delete ptr;
69 }
70 }
71
72 void setMode(Mode m) { mode = m; }
73 Mode getMode() const { return mode; }
74
75 private:
76 Mode mode;
77 };
78
79 /// @brief 观察者构建器
80 template<typename T>
81 auto makeObserver(T *ptr) {
82 return std::unique_ptr<T, SwitchDeleter<T> >(
83 ptr, SwitchDeleter<T>(SwitchDeleter<T>::Observe)
84 );
85 }

87 /// @brief 管理者构建器
88 template<typename T>
89 auto makeManage(T *ptr) {

93 }
