46 /// @brief 可控删除器
47 /// @details 可以为std::unique_ptr 提供一个删除器，使其可以有观察者与管理者两种状态
48 template<typename T>
49 class SwitchDeleter {
50 public:
51 enum Mode { Observe, Manage };
52
53 explicit SwitchDeleter(const Mode m = Manage) : mode(m) {
54 }
55
56 void operator()(const T *ptr) const noexcept {
57 if (mode == Manage) {
58 delete ptr;
59 }
60 }
61
62 void setMode(Mode m) { mode = m; }
63 Mode getMode() const { return mode; }
64
65 private:
66 Mode mode;
67 };
