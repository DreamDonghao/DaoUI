40 /// @brief 几何图形
41 /// @details 一个多边形
42 /// @tparam N 多边形边数

44 class Geometry {
45 public:
46 explicit Geometry(const std::array<GeometryVertex, N> &vertices): m_vertices(makeVertices(vertices)) {
47 int index = -1;
48 for (uint32 i = 1; i < N - 1; ++i) {
49 m_indices[++index] = 0;

51 m_indices[++index] = i + 1;
52 }
