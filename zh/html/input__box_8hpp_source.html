23 public:

29 /// @brief 处理消息，更新状态
30 void handleEvent(const SDL_Event &event) {
31 if (m_status == InputStatus::Disabled) {
32 return;
33 }
34 float x = 0, y = 0;
35 if (event.type == SDL_EVENT_MOUSE_MOTION) {
36 x = event.motion.x;
37 y = event.motion.y;
38 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN || event.type == SDL_EVENT_MOUSE_BUTTON_UP) {
39 x = event.button.x;
40 y = event.button.y;
41 }
42
43 SDL_Window *win = SDL_GetWindowFromID(event.window.windowID);
44 if (m_status == InputStatus::Input) {
45 const SDL_Rect rect{
46 static_cast<int>(m_boundingBox.getLeft()),
47 static_cast<int>(m_boundingBox.getTop()), // 顶部
48 static_cast<int>(m_boundingBox.getWidth()), // 宽度
49 static_cast<int>(m_boundingBox.getHeight()) // 高度
50 };
51 SDL_SetTextInputArea(win, &rect, cursorPos);
52 if (event.type == SDL_EVENT_TEXT_INPUT) {
53 const char *utf8chars = event.text.text;
54 const auto begin = utf8::unchecked::iterator(utf8chars);
55 const auto end = utf8::unchecked::iterator(utf8chars + std::strlen(utf8chars));
56
57 for (auto it = begin; it != end; ++it) {
58 m_text.insert(m_text.begin() + cursorPos, static_cast<char32_t>(*it));
59 cursorPos += 1;
60 }
61 } else if (event.type == SDL_EVENT_KEY_DOWN) {
62 if (event.key.key == SDLK_BACKSPACE && !m_text.empty()) {
63 if (cursorPos != 0) {
64 m_text.erase(cursorPos - 1, 1);
65 --cursorPos;
66 }
67 } else if (event.key.key == SDLK_DELETE) {
68 if (cursorPos != m_text.size()) {
69 m_text.erase(cursorPos, 1);
70 }
71 } else if (event.key.key == SDLK_LEFT) {
72 cursorPos = std::max(cursorPos - 1, 0);
73 } else if (event.key.key == SDLK_RIGHT) {
74 cursorPos = std::min(cursorPos + 1, static_cast<int>(m_text.size()));
75 }
76 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN && event.button.button == SDL_BUTTON_LEFT) {
77 if (!m_boundingBox.isInBoundingBox(x, y)) {
78 m_status = InputStatus::Normal;
79 SDL_StopTextInput(win);
80 }
81 }
82 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN && event.button.button == SDL_BUTTON_LEFT) {
83 if (m_boundingBox.isInBoundingBox(x, y)) {
84 m_status = InputStatus::Input;
85 SDL_StartTextInput(win);
86 }
87 }
88 }

110 int cursorPos = 0;
111 };
