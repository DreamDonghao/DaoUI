41 void Window::registerTexture(const uint32 &textureId) {
42 const AtlasRegion atlasRegion = getAtlasRegion(textureId);
43 // 加载新纹理图集
44 if (const uint32 atlasId = atlasRegion.atlasId; !m_atlasTextures.contains(atlasId)) {
45 const char *texturePath = atlasRegion.atlasPath;
46 m_atlasTextures[atlasId] = IMG_LoadTexture(m_renderer, texturePath);
47 SDL_SetTextureScaleMode(m_atlasTextures[atlasId], SDL_SCALEMODE_NEAREST);
48 if (m_atlasTextures[atlasId] == nullptr) {
49 std::cerr << "图集纹理加载失败! 路径: " << texturePath << " 错误: " << SDL_GetError() << std::endl;
50 m_atlasTextures.erase(atlasId);
51 }
52 }
