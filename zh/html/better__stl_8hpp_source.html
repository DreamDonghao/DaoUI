31 /// @brief 可控删除器
32 /// @details 可以为std::unique_ptr 提供一个删除器，使其可以有观察者与管理者两种状态
33 class SwitchDeleter {
34 public:
35 enum Mode { Observe, Manage };
36
37 explicit SwitchDeleter(const Mode m = Manage) : mode(m) {
38 }
39
40 void operator()(const T *ptr) const noexcept {
41 if (mode == Manage) {
42 delete ptr;
43 }
44 }
45
46 void setMode(Mode m) { mode = m; }
47 Mode getMode() const { return mode; }
48
49 private:
50 Mode mode;
51 };
52
53 /// @brief 观察者构建器
54 template<typename T>
55 auto makeObserver(T *ptr) {
56 return std::unique_ptr<T, SwitchDeleter<T> >(
57 ptr, SwitchDeleter<T>(SwitchDeleter<T>::Observe)
58 );
59 }
60
61 /// @brief 管理者构建器
62 template<typename T>
63 auto makeManage(T *ptr) {
64 return std::unique_ptr<T, SwitchDeleter<T> >(
65 ptr, SwitchDeleter<T>(SwitchDeleter<T>::Manage)
66 );
67 }
68}
