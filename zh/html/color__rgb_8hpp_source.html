27 constexpr uint8 hexByteToU8(const std::string_view sv) noexcept {
28 uint8 value = 0;
29 for (const char c: sv) {
30 value *= 16;
31 if (c >= '0' && c <= '9') value += c - '0';
32 else if (c >= 'A' && c <= 'F') value += c - 'A' + 10;
33 else if (c >= 'a' && c <= 'f') value += c - 'a' + 10;
34 else return 0;
35 }
36 return value;
37 }
38 /// @brief 将 hex 颜色格式转换为 RGBA 格式
39 constexpr ColorRGBA hexToRGBA(std::string_view hex) {
40 if (!hex.empty() && hex[0] == '#')
41 hex.remove_prefix(1);
42 const auto len = hex.size();
43 if (len != 6 && len != 8) {
44 return {255, 255, 255, 1.0f};
45 }
46 const uint8 r = hexByteToU8(hex.substr(0, 2));
47 const uint8 g = hexByteToU8(hex.substr(2, 2));
48 const uint8 b = hexByteToU8(hex.substr(4, 2));
49 const float32 a = (len == 8)? rgbNorm[hexByteToU8(hex.substr(6, 2))]: 1.0f;
50
51 if ((r == 0 && (hex[0] != '0' || hex[1] != '0')) ||
52 (g == 0 && (hex[2] != '0' || hex[3] != '0')) ||
53 (b == 0 && (hex[4] != '0' || hex[5] != '0')) ||
54 (len == 8 && hexByteToU8(hex.substr(6, 2)) == 0 && (hex[6] != '0' || hex[7] != '0'))) {
55 return {255, 255, 255, 1.0f};
56 }
57 return {r, g, b, a};
58 }
