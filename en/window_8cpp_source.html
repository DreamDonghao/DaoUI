33 void Window::registerTexture(const TextureEnum &textureName) {
34 const AtlasRegion atlasRegion = getAtlasRegion(textureName);
35 // 加载新纹理图集
36 if (const uint32_t atlasId = atlasRegion.atlasId; !atlasTextures.contains(atlasId)) {
37 const char *texturePath = atlasRegion.atlasPath;
38 atlasTextures[atlasId] = IMG_LoadTexture(m_renderer, texturePath);
39 SDL_SetTextureScaleMode(atlasTextures[atlasId], SDL_SCALEMODE_NEAREST);
40 if (atlasTextures[atlasId] == nullptr) {
41 std::cerr << "图集纹理加载失败! 路径: " << texturePath << " 错误: " << SDL_GetError() << std::endl;
42 atlasTextures.erase(atlasId);
43 }
44 }
